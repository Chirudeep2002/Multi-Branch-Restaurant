
-- 1. Create Database
CREATE DATABASE IF NOT EXISTS restaurant_monitoring_db;
USE restaurant_monitoring_db;

-- 2. Create Tables

-- Restaurant Table
CREATE TABLE Restaurant (
    restaurant_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    branding TEXT
);

-- Branch Table
CREATE TABLE Branch (
    branch_id INT AUTO_INCREMENT PRIMARY KEY,
    restaurant_id INT,
    location VARCHAR(255),
    manager_name VARCHAR(100),
    contact_info VARCHAR(100),
    FOREIGN KEY (restaurant_id) REFERENCES Restaurant(restaurant_id)
);

-- Employee Table
CREATE TABLE Employee (
    employee_id INT AUTO_INCREMENT PRIMARY KEY,
    branch_id INT,
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    role ENUM('Manager', 'Cashier', 'Chef', 'Chef_Helper', 'Waiter/Waitress'),
    payroll DECIMAL(10,2),
    performance_score DECIMAL(5,2),
    FOREIGN KEY (branch_id) REFERENCES Branch(branch_id)
);

-- Customer Table
CREATE TABLE Customer (
    customer_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    phone VARCHAR(15),
    email VARCHAR(100)
);

-- Product/Menu Items Table
CREATE TABLE Product (
    product_id INT AUTO_INCREMENT PRIMARY KEY,
    branch_id INT,
    name VARCHAR(100),
    description TEXT,
    price DECIMAL(10,2),
    status ENUM('Available', 'Out of Stock'),
    FOREIGN KEY (branch_id) REFERENCES Branch(branch_id)
);

-- Order Table
CREATE TABLE Orders (
    order_id INT AUTO_INCREMENT PRIMARY KEY,
    customer_id INT,
    branch_id INT,
    order_time DATETIME,
    order_mode ENUM('Dine-in', 'Takeout', 'Delivery'),
    FOREIGN KEY (customer_id) REFERENCES Customer(customer_id),
    FOREIGN KEY (branch_id) REFERENCES Branch(branch_id)
);

-- Order_Items Table (many-to-many relationship)
CREATE TABLE Order_Items (
    order_id INT,
    product_id INT,
    quantity INT,
    PRIMARY KEY(order_id, product_id),
    FOREIGN KEY (order_id) REFERENCES Orders(order_id),
    FOREIGN KEY (product_id) REFERENCES Product(product_id)
);

-- Payments Table
CREATE TABLE Payments (
    payment_id INT AUTO_INCREMENT PRIMARY KEY,
    order_id INT,
    amount DECIMAL(10,2),
    method ENUM('Cash', 'Card', 'Online'),
    payment_time DATETIME,
    FOREIGN KEY (order_id) REFERENCES Orders(order_id)
);

-- Inventory Table
CREATE TABLE Inventory (
    inventory_id INT AUTO_INCREMENT PRIMARY KEY,
    branch_id INT,
    product_id INT,
    quantity_available INT,
    threshold_level INT,
    FOREIGN KEY (branch_id) REFERENCES Branch(branch_id),
    FOREIGN KEY (product_id) REFERENCES Product(product_id)
);

-- Supplier Table
CREATE TABLE Supplier (
    supplier_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    contact_info VARCHAR(100)
);

-- Supplier Inventory Supply Table
CREATE TABLE Supplier_Inventory (
    supplier_id INT,
    inventory_id INT,
    supply_date DATE,
    quantity_supplied INT,
    invoice_number VARCHAR(50),
    PRIMARY KEY (supplier_id, inventory_id),
    FOREIGN KEY (supplier_id) REFERENCES Supplier(supplier_id),
    FOREIGN KEY (inventory_id) REFERENCES Inventory(inventory_id)
);

-- Feedback Table
CREATE TABLE Feedback (
    feedback_id INT AUTO_INCREMENT PRIMARY KEY,
    customer_id INT,
    order_id INT,
    rating INT,
    comment TEXT,
    feedback_date DATE,
    FOREIGN KEY (customer_id) REFERENCES Customer(customer_id),
    FOREIGN KEY (order_id) REFERENCES Orders(order_id)
);

-- Government Table
CREATE TABLE Government (
    gov_id INT AUTO_INCREMENT PRIMARY KEY,
    department_name VARCHAR(100),
    contact_info VARCHAR(100)
);

-- Branch Government Transactions
CREATE TABLE Branch_Gov_Transaction (
    transaction_id INT AUTO_INCREMENT PRIMARY KEY,
    branch_id INT,
    gov_id INT,
    payment_amount DECIMAL(10,2),
    transaction_date DATE,
    FOREIGN KEY (branch_id) REFERENCES Branch(branch_id),
    FOREIGN KEY (gov_id) REFERENCES Government(gov_id)
);

-- Third Party Services Table
CREATE TABLE ThirdPartyService (
    service_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    service_type VARCHAR(50),
    contact_info VARCHAR(100)
);

-- Branch Third Party Service Table
CREATE TABLE Branch_Service (
    branch_id INT,
    service_id INT,
    cost DECIMAL(10,2),
    service_date DATE,
    PRIMARY KEY (branch_id, service_id),
    FOREIGN KEY (branch_id) REFERENCES Branch(branch_id),
    FOREIGN KEY (service_id) REFERENCES ThirdPartyService(service_id)
);

-- User/Admin Table
CREATE TABLE Users (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) UNIQUE,
    password_hash VARCHAR(255),
    role ENUM('Admin', 'Branch_Manager', 'Viewer'),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Reports Table
CREATE TABLE Reports (
    report_id INT AUTO_INCREMENT PRIMARY KEY,
    branch_id INT,
    report_type ENUM('Sales', 'Inventory', 'Performance'),
    report_date DATE,
    content TEXT,
    FOREIGN KEY (branch_id) REFERENCES Branch(branch_id)
);

-- Performance Metrics Table
CREATE TABLE PerformanceMetrics (
    metric_id INT AUTO_INCREMENT PRIMARY KEY,
    branch_id INT,
    month_year VARCHAR(7), -- Format 'YYYY-MM'
    sales_volume DECIMAL(10,2),
    customer_count INT,
    avg_feedback_score DECIMAL(4,2),
    FOREIGN KEY (branch_id) REFERENCES Branch(branch_id)
);

-- Login Logs Table
CREATE TABLE LoginLogs (
    log_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    login_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    action TEXT,
    FOREIGN KEY (user_id) REFERENCES Users(user_id)
);
